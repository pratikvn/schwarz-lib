123 template <typename ValueType, typename IndexType>
124 void transfer_buffer(const Settings &settings, MPI_Win &window,
125  ValueType *target_buffer, IndexType **num_elems,
126  int offset, int target_subd, IndexType *neighbors,
127  IndexType *displacements)
128 {
129  ValueType dummy = 1.0;
130  auto mpi_vtype = boost::mpi::get_mpi_datatype(dummy);
131  if (settings.comm_settings.enable_lock_local) {
132  MPI_Win_lock(MPI_LOCK_SHARED, neighbors[target_subd], 0, window);
133  }
134  if (settings.comm_settings.enable_put) {
135  MPI_Put(&target_buffer[offset], (num_elems[target_subd])[0], mpi_vtype,
136  neighbors[target_subd], displacements[neighbors[target_subd]],
137  (num_elems[target_subd])[0], mpi_vtype, window);
138  } else if (settings.comm_settings.enable_get) {
139  MPI_Get(&target_buffer[offset], (num_elems[target_subd])[0], mpi_vtype,
140  neighbors[target_subd], displacements[neighbors[target_subd]],
141  (num_elems[target_subd])[0], mpi_vtype, window);
142  }
143  if (settings.comm_settings.enable_flush_all) {
144  MPI_Win_flush(neighbors[target_subd], window);
145  } else if (settings.comm_settings.enable_flush_local) {
146  MPI_Win_flush_local(neighbors[target_subd], window);
147  }
148  if (settings.comm_settings.enable_lock_local) {
149  MPI_Win_unlock(neighbors[target_subd], window);
150  }
151 }
